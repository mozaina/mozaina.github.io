Linux下使用shell批量修改文件名的三种方法

2020年9月2日
| 阅读

1、通过sed替换

[root@node01 wolf]# find / -type f -name wolf.log
/wolf.log
/tmp/wolf.log
/root/wolf/wolf.log
[root@node01 wolf]# sed -i ’s#wolf#yujing#g’ find / -type f -name wolf.log
[root@node01 wolf]# find / -type f -name “wolf.log” | xargs cat

yujing
yujing
yujing
[root@node01 wolf]#
