数据库备份？

#!/bin/bash
TABASE_MESSAGE=`mysql -uroot -predhat -EN -e "show databases;" | grep -E "^\*|schema$" -v`
mkdir -p /mnt/mysql_dump
for DATABASE_NAME in $DATABASE_MESSAGE
do
        mysqldump -uroot $DATABASE_NAME > /mnt/mysql_dump/${DATABASE_NAME}.sql
        [ "$?" -eq "0" ]&&{
                echo -e "\033[32m$DATABASE_NAME is backuped !!\033[0m"
        }
done

根分区容量超过10%时给root用户发邮件报警？

#!/bin/bash
Use=`df | awk -F " " '/\/$/{print $5}'| awk -F "%" '{print $1}'`
while true
do
        [ "$Use" -ge "10" ]&&{
                echo "your / will full" | mail -s warning root
        }
        sleep 1
done


根分区容量超过10%时将报警显示在当前shell上？

#!/bin/bash
Use=`df | awk -F " " '/\/$/{print $5}' | awk -F "%" '{print $1}'`
TTY=`ps | awk '/bash$/{print $2}'`
while true
do
        [ "$Use" -ge 10 ]&&{
                echo "your / will full !!" > /dev/$TTY
        }
        sleep 5
done
