8Shell编程从入门到精通文本处理利器awk

2020年9月2日
| 阅读 
//用 --- 把1，3连起来
awk '{print $1 "---" $3}' /etc/fstab

awk '{print "device:" $1 "\t" $3}' /etc/fstab

//多个分隔符，可以是 \t 也可以是空格
awk -F"[\t ]+" '{print "device:" $1 "\t FSTYPE : " $3}' /etc/fstab

//匹配空行
awk '/^$/ {print "This is an empty line"}' /etc/inittab

//匹配 bash
awk -F":" '/bash/ {print $0}' /etc/passwd
