01 #!/bin/sh
02 myPath="/var/log/httpd/"
03 myFile="/var /log/httpd/access.log"
04 # 这里的-x 参数判断$myPath是否存在并且是否具有可执行权限
05 if [ ! -x"$myPath"];then
06 mkdir "$myPath"
07 fi
08 # 这里的-d 参数判断$myPath是否存在
09 if [ ! -d"$myPath"];then
10 mkdir "$myPath"
11 fi
12
13 # 这里的-f参数判断$myFile是否存在
14 if [ ! -f"$myFile" ];then
15 touch "$myFile"
16 fi
