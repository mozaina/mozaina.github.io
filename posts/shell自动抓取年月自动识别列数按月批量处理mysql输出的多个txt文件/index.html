#!/bin/bash
bash_dir=“/root/bash/tongji” # 脚本存放目录
txt_files=“$bash_dir/txt” # 要被统计的txt文件存放目录
re=”[0-9]{4}-[0-9]{2}” # 被统计的txt文件第一列是日期，而且格式为”XXXX-XX-XX”
start_year=$(cat $txt_files/ | gawk ‘$1~/[0-9]{4}-[0-9]{2}/ {print $1}’ | gawk -F- ‘{print $1}’ | sort -u | head -n1)
end_year=$(cat $txt_files/ | gawk ‘$1~/[0-9]{4}-[0-9]{2}/ {print $1}’ | gawk -F- ‘{print $1}’ | sort -u | tail -n1)
start_month=$(cat $txt_files/ | gawk ‘$1~/[0-9]{4}-[0-9]{2}/ {print $1}’ | gawk -F- ‘{print $2}’ | sort -u | head -n1)
end_month=$(cat $txt_files/ | gawk ‘$1~/[0-9]{4}-[0-9]{2}/ {print $1}’ | gawk -F- ‘{print $2}’ | sort -u | tail -n1)
start_day=“01”
end_day=$(echo cal $end_month $end_year | gawk ‘{print $NF}’)
start_date=“$start_year-$start_month-$start_day”
end_date=“$end_year-$end_month-$end_day”
start_dateSec=$(date -d “-0 day $start_date” +%s)
end_dateSec=$(date -d “-0 day $end_date” +%s)
title=(cat $txt_files/* | grep -Ev "($re)" | head -n1)
default_name=$(ls $txt_files | gawk -F. ‘{print $1}’ | head -n1)
echo
echo -en “\033[33m ==> Please input new file’s name what you want, default name is\033[35m [ $default_name ]\033[0m : “
read filename
if [ ! $filename ];then
filename=$default_name
fi
data_file=“$bash_dir/alldata/${filename}${start_date}to${end_date}.txt”
echo
rm -rf /tmp/date_list.txt
rm -rf $data_file
for line in ${title[]}
do
echo -en “$line\t” >> $data_file
done
echo >> $data_file
for i in $(seq 1 1000)
do
until [ $start_dateSec -eq $end_dateSec ]
do
if [ $start_dateSec -ge $end_dateSec ];then
echo $end_date >> /tmp/date_list.txt
end_date=$(date -d “+1 day $end_date” +%F)
end_dateSec=$(date -d $end_date +%s)

end_dateSec=$(date -d “-1 day $end_date” +%s)
elif [ $start_dateSec -le $end_dateSec ];then
echo $start_date >> /tmp/date_list.txt
start_date=$(date -d “+1 day $start_date” +%F)
start_dateSec=$(date -d $start_date +%s)
start_dateSec=$(date -d “-1 day $start_date” +%s)
else
break
fi
done
done

for date in cat /tmp/date_list.txt
do
for num in seq 2 ${#title[@]}
do
eval sum_${num}=0
done
echo -en “$date\t” >> $data_file
for name in cat $txt_files/* | grep -E "($re)" | sed 's/\t/,/g'
do
L_1=$(echo $name | gawk -F, ‘{print $1}’)
if [ “$L1” == $date ];then
for num in seq 2 ${#title[@]}
do
eval L${num}=$(echo $name | gawk -F, -vi=$num ‘{print $i}’)
eval let sum$num=‘$‘sum”$num”+‘$‘L”$num”
done
fi
done
for num in seq 2 ${#title[@]}
do
eval echo -en ‘$‘sum”$num” >> $data_file
echo -en “\t” >> $data_file
done
echo >> $data_file
echo -e “\033[32m [ O K ] The \033[35m$date\033[32m date count finished……\033[0m”
done

rm -rf /tmp/date_list.txt
echo
sz $data_file
