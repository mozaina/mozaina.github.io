mongodb shell 运行js 脚本

实战

编写js脚本

编写一段mongodb的跨库联查的统计语句，如下：

//连接到media库
var db1 = connect("media");
var widgetArray = db1.widget.find({ }, { "identify" : 1 });
//链接到monitor
var db2 = connect("monitor");
//find得到的都是迭代器
while(widgetArray.hasNext()){
    var temp = widgetArray.next();
    var count = db2.h5monitor.find({"identifier": temp.identify,"create_time":{"$lt":new Date("2015-11-09 23:59:59"),"$gte":new Date("2015-10-01 0:0:0")},"action":"LOGIN"}).count();
    var count2 = db2.runCommand({distinct:'h5monitor',key:'username',query:{"identifier": temp.identify,"create_time":{"$lt":new Date("2015-11-09 23:59:59"),"$gte":new Date("2015-10-01 0:0:0")},"action":"LOGIN"}}).values.length;
    var result = temp.identify+","+count+","+count2;
    //打印结果
    print(result);
}

在linux中跑脚本

#./mongo 127.0.0.1:27017 /home/soft/hehe.js
  • 会在console界面打印结果

在linux中跑后台脚本

#nohup ./mongo 127.0.0.1:27017 /home/soft/hehe.js > myresult.file

  • 使用linux命令[nohup][5]

  • 把结果输出到[myresult.file][5]中，使用[>][5]管道流重定向
