函数式编程

# Lambda
sum  = ->(x, y) { x + y }
diff = ->(x, y) { x - y }

def double(x, y, function)
  2 * function.call(x, y)
end

double(5, 10, sum)  #=> Output : 2 * (5+10) = 30
double(2, 1, diff)  #=> Output:  2 * (2-1) = 2
