#!/bin/bash
set -x

function rm OldRelease(){
    awk -F "/" '{print $13}' release.log >> tmp.txt

    while read line || [[ -n $line ]]
    do
        if [ ${line} ];then
            rm -rf ${line}
            sed -i "/${line}/d" release.log
        fi
    done < tmp.txt

    if [ -f tmp.txt ];then
        rm -f tmp.txt
    fi

    sed -i '$!d' release.log
}

ls -l |awk '{print $9}' |grep -v "^$" >dirList.txt

while read line || [[ -n $line ]]
do
    if [[ ${line} =~ ^HiCloud.*$ ]];then
        pushd /opt/hw/release/hitouch/${line}
        rmOldRelease
        popd
    fi
done < dirList.txt

rm -f dirList.txt
