安装一个ruby版本

# source /etc/profile.d/rvm.sh
# rvm list known
# rvm install 1.9.3
# rvm use 1.9.3
# rvm remove 1.9.3

# rvm install 2.4.4
# rvm use 2.4.4
# ruby --version
ruby 2.3.4p301 (2017-03-30 revision 58214) [x86_64-linux]
