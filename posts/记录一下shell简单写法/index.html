记录一下shell简单写法

2020年9月2日
| 阅读

#!/bin/bash

DT=date +%Y%m%d #当前日期
YESTERDAY=date -d "yesterday" +%Y-%m-%d #昨天，用于处理数据的日期

if [ -z “$1”]; then
echo “ETL Date:$YESTERDAY”;
else
YESTERDAY=${1}
echo “ETL Date:$YESTERDAY”;
fi

QUY_HOST=
QUY_DB=
QUY_USER=
QUY_PWD=

#导出服务分类
mysql -h$QUY_HOST $QUY_DB -u$QUY_USER -p$QUY_PWD –default-character-set=utf8 –skip-column-names -B -e “SELECT dictionary_key,dictionary_value FROM sys_dictionary WHERE module=‘QUERY_SCENE’” > etl_servicecategory$DT.txt

RES_HOST=
RES_USER=
RES_PWD=

#导出调用次数
mysql -h$RES_HOST -u$RES_USER -p$RES_PWD –default-character-set=utf8 –skip- column-names -B -e “SELECT DATE(FROM_UNIXTIME(created_at)) query_date,device_type,service_id,if(res_db=“,NULL,res_db) res_db,COUNT(*) FROM roobochat.rc_chat_records WHERE created_at BETWEEN UNIX_TIMESTAMP(‘$YESTERDAY 00:00:00’) AND UNIX_TIMESTAMP(‘$YESTERDAY 23:59:59’) AND response like ‘http%’ GROUP BY query_date,device_type,service_id,resdb” > etl$DT.txt

echo ‘SQL data file is created.’

ECP_HOST=
ECP_USER=
ECP_PWD=
ECP_DATABASE=prod_ecp_db

#写入到统计表
mysql -h$ECP_HOST $ECP_DATABASE -u$ECP_USER -p$ECP_PWD << EOF #将“内容段”整个作为命令的输入。
LOAD DATA LOCAL INFILE ‘etl_servicecategory$DT.txt’ INTO TABLE tp_query_service_category;
DELETE FROM ts_query_voice WHERE querydate = DATE(‘$YESTERDAY’);
LOAD DATA LOCAL INFILE ‘./etl$DT.txt’ INTO TABLE ts_query_voice(query_date,device_type,service_id,source,total);
UPDATE ts_query_voice t1 INNER JOIN tp_query_service_category t2 ON t1.service_id = t2.service_id SET t1.service_name = t2.service_name WHERE t1.query_date = DATE(‘$YESTERDAY’);
EOF

echo ‘ETL succeeded.’

crontab -e

0 3 * * * /home/edw/etl_sh/query_voice.sh
