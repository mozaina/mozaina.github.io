Android服务器环境apk打包shell脚本

2020年9月2日
| 阅读

背景：18年底入职一家创业4年的公司，除去丢弃的一款购物类产品，现在主要运营一款数据量增长较快的工具类产品，前景可观，尤其在19年初微商法出台后，竞争尤其激烈，刚入职主要负责测试管理及性能测试，完善研发

测试流程，从开发手里接过测试环境，连终端App打包服务器，都可以自己打包，即使开发在jenkins集成自动打包发包，android打包工具gradlew：

#!/bin/sh
read -p "请输入所打包的分支：" branch

apk_dir="/home/devtest/android/android_project/项目工程/dragon/build/outputs/apk"

log_dir="/home/devtest/logs"

log_file="$log_dir/apk.log"

log_aR_file="$log_dir/aR_apk.log"

build_dir="/home/devtest/android/android_project/项目工程/"

cd $build_dir

cpwd=`pwd`

if [ "$build_dir" == "$cpwd" ];then
    echo "提取打包分支，与请求的打包分支是否一致"
    git checkout $branch
else
    branch_local=`git branch -a |grep $branch |awk '{print $2}' |head -1`
    echo "当前打包分支：$branch_check"
fi

branch_check=`git branch -a |grep $branch |awk '{print $2}' |head -1`

echo "当前路径：";pwd

if [ "$branch_check"  == "$branch" ]
then
    echo "实际打包分支：$branch"
    git pull &>> $log_file ;tail  -n 10 $log_file
    echo "成功拉取代码！" 
else 
    echo "实际打包分支：$branch"
    echo "打包分支不正确，exit执行当前脚本"
fi

./gradlew aR  &>> $log_aR_file ;tailf $log_aR_file

if [ -z $apk_dir ]
then
    echo "取包路径$apk_dir 不存在"
else
    echo "apk取包路径:$apk_dir"
fi
