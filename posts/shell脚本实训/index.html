#!/bin/bash
fir=0
sec=2
temp=0
lines=$(cut -d ' ' -f1 userInfo.txt | wc -l)
while [ $temp -le $lines ]
do
temp=$((fir+sec))
fir=${sec}
sec=${temp}
echo "$temp," $(sed -n "${temp}p" userInfo.txt) >> result.txt
done

要求：

（1）取出2， 4， 6， 10， 16……等行的用户信息

（2）并将结果保存到result.txt中
