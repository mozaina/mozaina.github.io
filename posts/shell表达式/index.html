1.算术表达式

let 算术运算表达式

let c= a + a+ a+b

$[算术表达式] 不常用

c= [ [ [a+$b]

$((算术表达式))

c= ( ( (( ((a+ b ) ) 或 者 c = b))或者c= b))或者c=((a+b))

通过type let命令，可以知道let命令是一个shell builtin，因此可以用help命令查看：
let是一个算术表达式，里面有一些计算公式：

[root@node1 ~]# a=1
[root@node1 ~]# b=1
[root@node1 ~]# let c=$a+$b
[root@node1 ~]# echo $c
2
[root@node1 ~]# c=$[$a+$b]
[root@node1 ~]# c=$(($a+$b))
