crontab定时任务执行找不到库orshell可执行crontab定时任务下就不能执行tensorflowImportErrorlibcudaso1cannotopen

2020年9月2日
| 阅读

在线上启动一个定时任务，但是起来查看，发现任务执行找不到库，报cuda错误：

ImportError: libcuda.so.1: cannot open shared object file: No such file or directory

后面在/usr/local/nvidia/lib64 找到了该库，nvidia- smi也没问题，那应该就是变量的问题，参考了网上（https://www.cnblogs.com/shenfeng/p/crontab_fail_executing.html）的一个比较暴力的方案：

cat /etc/crontab   
可以发现，有些路径还是没有加进去的，可以通过

可以先通过export -p命令获得正常执行时该用户下的环境变量，然后复制这些环境变量到需要执行的脚本中。

declare -x HISTSIZE="1000"
declare -x HOME="/home/oracle"
declare -x HOSTNAME="test-rac2.test.net"
declare -x LANG="en_US.UTF-8"  
 **declare -x LD_LIBRARY_PATH="/usr/local/nvidia/lib: /usr/local/nvidia/lib64:/usr/local/cuda/lib64:"**
declare -x LESSOPEN="|/usr/bin/lesspipe.sh %s"
declare -x LOGNAME="oracle"
declare -x LS_COLORS="rs=0:di=01;34:ln=01;36:mh=00:pi=40;33:so=01;35:do=01;35:bd=40;33;01:cd=40;33;01:or=40;31;01:mi=01;05;37;41:su=37;41:sg=30;43:ca=30;41:tw=30;42:ow=34;42:st=37;44:ex=01;32:*.tar=01;31:*.tgz=01;31:*.arj=01;31:*.taz=01;31:*.lzh=01;31:*.lzma=01;31:*.tlz=01;31:*.txz=01;31:*.zip=01;31:*.z=01;31:*.Z=01;31:*.dz=01;31:*.gz=01;31:*.lz=01;31:*.xz=01;31:*.bz2=01;31:*.tbz=01;31:*.tbz2=01;31:*.bz=01;31:*.tz=01;31:*.deb=01;31:*.rpm=01;31:*.jar=01;31:*.rar=01;31:*.ace=01;31:*.zoo=01;31:*.cpio=01;31:*.7z=01;31:*.rz=01;31:*.jpg=01;35:*.jpeg=01;35:*.gif=01;35:*.bmp=01;35:*.pbm=01;35:*.pgm=01;35:*.ppm=01;35:*.tga=01;35:*.xbm=01;35:*.xpm=01;35:*.tif=01;35:*.tiff=01;35:*.png=01;35:*.svg=01;35:*.svgz=01;35:*.mng=01;35:*.pcx=01;35:*.mov=01;35:*.mpg=01;35:*.mpeg=01;35:*.m2v=01;35:*.mkv=01;35:*.ogm=01;35:*.mp4=01;35:*.m4v=01;35:*.mp4v=01;35:*.vob=01;35:*.qt=01;35:*.nuv=01;35:*.wmv=01;35:*.asf=01;35:*.rm=01;35:*.rmvb=01;35:*.flc=01;35:*.avi=01;35:*.fli=01;35:*.flv=01;35:*.gl=01;35:*.dl=01;35:*.xcf=01;35:*.xwd=01;35:*.yuv=01;35:*.cgm=01;35:*.emf=01;35:*.axv=01;35:*.anx=01;35:*.ogv=01;35:*.ogx=01;35:*.aac=01;36:*.au=01;36:*.flac=01;36:*.mid=01;36:*.midi=01;36:*.mka=01;36:*.mp3=01;36:*.mpc=01;36:*.ogg=01;36:*.ra=01;36:*.wav=01;36:*.axa=01;36:*.oga=01;36:*.spx=01;36:*.xspf=01;36:" declare -x MAIL="/var/spool/mail/oracle" declare -x OLDPWD declare -x ORACLE_BASE="/u1/app/oracle" declare -x ORACLE_HOME="/u1/app/oracle/product/12.1.0/db1" declare -x ORACLE_SID="TEST2"
