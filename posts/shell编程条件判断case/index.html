#!/bin/bash
echo " 1   |   fengjie    "
echo " 2   |   canglaoshi "
echo " 3   |   boduoye    "
read -p "please input your choice:   " -t 30 girl
case "$girl" in
        "1")
                echo "your choice is fengjie!!"
                ;;
        "2")
                echo "your choice is canglaoshi!!!"
                ;;
        "3")
                echo "your choice is doduoye!!!"
                ;;
        *)
                echo "error input !!"
                ;;
esac
